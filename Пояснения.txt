from flask import Flask

#Создание объекта
app = Flask(__name__)# передача названия файла  при помощи директивы __name__
# конструктор класса

#Отслеживание URL ажресса и 
# в зависимости от адресса показываем ту или иную информацию

#Чтобы отследить какой либо адресс создаем декоратор
# '/' - отслеживание главвной страницы
# то есть при переходе на главную строничку я хочу выводить какой то текст
# то есть здесь мы обрабатываем главную строничку
@app.route('/')
#@app.route('/home') отслеживаем 2 адресса
# выводим текст
def index():
    return "Hello world"

# обрабатвыаем еще один url адресс
@app.route('/about')
def about():
    return "About page"
